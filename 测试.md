processCode(target) {
		let result = [];
		let text = target.html();
		let doc = $(text);//this.domParser.parseFromString(text, 'text/xml');
		$(doc).filter((index, value) => {return value.nodeType === 1;}).each((ignore, element) => {
			let target = $(element);
			let type = target.attr('lang');
			let code = $.trim(target.html());
			result.push({type, code});
		});
		return result;
	}
