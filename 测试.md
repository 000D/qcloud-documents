<div lang="C++">
processCode(target) {
		let result = [];
		let text = target.html();
		let doc = $(text);//this.domParser.parseFromString(text, 'text/xml');
		$(doc).filter((index, value) => {return value.nodeType === 1;}).each((ignore, element) => {
			let target = $(element);
			let type = target.attr('lang');
			let code = $.trim(target.html());
			result.push({type, code});
		});
		return result;
	}
</div>



<div lang="A++">
renderCode({data, target}) {
		let codeTemplate = $(this.codeTemplate);//$(this.domParser.parseFromString(this.codeTemplate, 'text/xml'));
		let langList = '';
		let codeList = '';
		let className;
		let styles;
		$(data).each((index, obj) => {
			className = index > 0 ? '' : 'active';
			styles = index > 0 ? 'style = "display: none"' : '';
			langList += `<li class="${className}"><a class="J-langName" data-bypass-xhr="1" href="javascript: void 0;" data-type="_${obj.type}">${obj.type}</a></li>`;
			codeList += `<code ${styles} data-type="_${obj.type}">${obj.code}</code>`;
		})
		codeTemplate.find('.J-language').html(langList);
		codeTemplate.find('.J-code').html(codeList);
		codeTemplate.insertAfter(target);
	}
</div>
</script>
