function processScript(script, style) {
				var current = script;
				while (current = current.nextSibling) {
					//过滤文本节点
					if (current.nodeType != 1) {
						continue;
					}
					if (current.getAttribute('action') === 'end' 
					 || current.getAttribute('type') === 'text/plugin'
					) {
						break;
					}
					current.style.display = style || 'none';
				}
			}
